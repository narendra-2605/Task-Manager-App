import React, { useState } from "react";
import DisplayTask from "./DispalyTask";

/**
 * 
 * @param {props} addTask function as a props
 */
const AddTask = ({ addTask }) => {
    /**
     * setUserInput state for storing the task in the userState state
     */
    const [userInput, setUserInput] = useState('');
    /**
     * handleSubmit function will execute for adding the new task
     * @param {string} e - for getting the vaue of the event used in the handleSubmit function
     */
    const handleSubmit = (e) => {
        /**
         * for preventing the blank input
         */
        if (!userInput.length) {
            alert("Please input somthing");
        }
        /**
         * preventDefault is used to stop the reloading the page again and again
         */
        e.preventDefault();
        addTask(userInput);
        setUserInput("");
        /**
         * This code is used to clear the input field after submiting the form once
         */
        Array.from(document.querySelectorAll("input")).forEach(
            input => (input.value = "")
          );
    }

    return (
        <>
            <div className="input-group mb-3">
                <input className="form-control" placeholder="Enter The Task" aria-label="Recipient's username" aria-describedby="basic-addon2" onChange={(e) => { setUserInput(e.target.value) }} type="text" />
                <div className="input-group-append">
                    <button className="input-group-text" id="basic-addon2" onClick={handleSubmit}> Add Task</button>
                </div>
            </div>
        </>
    );
}

export default AddTask
