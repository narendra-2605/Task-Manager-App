import React, { useState } from "react";
import ToDo from "./ToDo";
// import demoTodo from './demoTodo.json'

/**
 * 
 * @param {props} state and function as a props
 */
const DisplayTask = ({ toDoList, handleToggle, handleFilter }) => {
    /**
     * display function is return the ToDo component
     * @returns the Todo component for diplaying the tasks
     */
    const display = () => {
        return (
            <>
                {toDoList?.map(todo => {
                    return (
                        <ToDo todo={todo} handleToggle={handleToggle} handleFilter={handleFilter} />
                    )
                })}
            </>
        )
    }

    return (
        <>
            {
                <div>
                    {display()}
                    {/* <button style={{ margin: '20px' }} onClick={handleFilter}>Clear Completed</button> */}
                </div>
            }
        </>
    );
}

export default DisplayTask
