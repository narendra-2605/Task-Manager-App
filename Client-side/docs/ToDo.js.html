import React from 'react';
/**
 * 
 * @param {props} todo and handleToggle function as a props
 */

const ToDo = ({ todo, handleToggle }) => {

    /**
     * handleClick function will call the handleToggle function 
     * @param {string} e  The event for storing the id 
     */
    const handleClick = (e) => {
        /**
         * preventDefault is used to stop the reloading the page again and again
         */
        e.preventDefault()
        /**
        * Represents the handleToggle function it will map the task for identifying it as completed or not
        * @param {number} id - The Id of the task
        */
        handleToggle(e.currentTarget.id)
    }

    return (
        <p id={todo.id} key={todo.id + todo.task} name="todo" value={todo.id} onClick={handleClick} style={{ backgroundColor: todo.complete ? "red" : "#13e421", cursor: "pointer" }}> {todo.task}</p>
        // <input className="mr-1" type="checkbox" onChange={handleClick}  />  
    );
}

export default ToDo
